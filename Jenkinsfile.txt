pipeline {
    agent any
    
    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/ranya-hamzaoui/testjenkins.git'
            }
        }
        stage('install Dependances') {
            steps {
                script {
                    sh 'npm install'
                }
            }
        }
        stage('Build') {
            steps {
                script {
                    sh 'npm run build'
                }
            }
        }
        stage('Build Docker Image') {
            steps {
                script {
                    sh 'docker build -t raniahamzaoui91/angular-app .'
                }
            }
        }
        stage('Push Docker Image') {
            steps {
                script {
                    sh 'docker push raniahamzaoui91/angular-app'
                }
            }
        }
        stage('Deploy to Kubernetes') {
            steps {
                script {
                    sh 'kubectl apply -f k8s/deployment.yaml'
                }
            }
        }
    }
}
